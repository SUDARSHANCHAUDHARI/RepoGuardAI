# Daily Security Automation Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Package RepoGuardAI for GitHub Actions and add free daily dependency, code, and secret scanning with reviewable dependency PRs and deduplicated high-risk reporting.

**Architecture:** RepoGuardAI supplies a bundled JavaScript Action and a reusable workflow. The reusable workflow runs untrusted scans with read-only permissions, then hands generated artifacts to separate SARIF and issue-reporting jobs with narrowly scoped permissions. Dependabot owns dependency-update PRs; RepoGuardAI never auto-merges or rewrites application code.

**Tech Stack:** TypeScript 5, Node.js 18+ CLI, Node.js 24 GitHub Action runtime, Commander, Zod, Vitest, tsup, pnpm, GitHub Actions, Dependabot, CodeQL, Semgrep CE 1.169.0, OSV Scanner 2.3.8, Gitleaks Action 3.0.0.

## Global Constraints

- Use only free GitHub features and open-source scanners.
- Keep Node.js 18 compatibility for the published CLI; only the GitHub Action entrypoint uses Node.js 24.
- Run pull-request scans with `contents: read`, no repository secrets, and no write operations.
- Isolate `security-events: write` and `issues: write` in jobs that do not execute target repository code.
- Do not use `pull_request_target`.
- Do not automatically merge dependency pull requests.
- Do not automatically rewrite application code for security findings.
- Do not print or publish detected secret values.
- Pin every third-party GitHub Action to a verified immutable commit SHA with a release comment.
- Preserve reports through `if: always()` before applying the final failure result.
- Stage files by explicit path and keep every commit independently reviewable.

- [ ] **Step 3: Implement the minimal gate module**

Create `src/severity-gate.ts`:

```ts
import { SEVERITIES } from "./schemas.js";
import type { Finding, Severity } from "./types.js";

export const FAILURE_THRESHOLDS = [...SEVERITIES, "none"] as const;
export type FailureThreshold = (typeof FAILURE_THRESHOLDS)[number];

const rank = new Map<Severity, number>(
  SEVERITIES.map((severity, index) => [severity, index]),
);

export function parseFailureThreshold(value: string): FailureThreshold {
  const normalized = value.toLowerCase();
  if (!FAILURE_THRESHOLDS.includes(normalized as FailureThreshold)) {
    throw new Error(`Unknown failure threshold "${value}". Use: ${FAILURE_THRESHOLDS.join(", ")}.`);
  }
  return normalized as FailureThreshold;
}

export function findingsAtOrAbove(
  findings: Finding[],
  threshold: FailureThreshold,
): Finding[] {
  if (threshold === "none") return [];
  const thresholdRank = rank.get(threshold) ?? Number.NEGATIVE_INFINITY;
  return findings.filter(
    (finding) =>
      finding.status !== "rejected" &&
      (rank.get(finding.severity) ?? Number.POSITIVE_INFINITY) <= thresholdRank,
  );
}

export function shouldFailAudit(
  findings: Finding[],
  threshold: FailureThreshold,
): boolean {
  return findingsAtOrAbove(findings, threshold).length > 0;
}
```

  findingsAtOrAbove,
  parseFailureThreshold,
} from "./severity-gate.js";

type AuditOptions = {
  security?: boolean;
  api?: boolean;
  failOn?: string;
};

// On the audit command:
.option(
  "--fail-on <severity>",
  "exit 2 after reports when active findings meet critical|high|medium|low|informational|none",
)
.action((repository: string, opts: AuditOptions) => {
  // existing audit pipeline remains unchanged through writeReports(...)
  if (opts.failOn) {
    let threshold;
    try {
      threshold = parseFailureThreshold(opts.failOn);
    } catch (error) {
      fail(error instanceof Error ? error.message : String(error));
    }
    const blocking = findingsAtOrAbove(findings, threshold);
    if (blocking.length > 0) {
      console.error(
        `✖ ${blocking.length} active finding(s) meet the ${threshold} failure threshold.`,
      );
      process.exitCode = 2;
    }
  }
});

export interface ActionInputs {
  target: string;
  scope: "full" | "security" | "api";
  failOn: string;
}

export function buildAuditArgs(inputs: ActionInputs): string[] {
  if (!(["full", "security", "api"] as const).includes(inputs.scope)) {
    throw new Error("scope must be one of: full, security, api");
  }
  const args = ["audit", inputs.target];
  if (inputs.scope !== "full") args.push(`--${inputs.scope}`);
  if (inputs.failOn) args.push("--fail-on", inputs.failOn);
  return args;
}

- [ ] **Step 3: Implement pure filtering and rendering helpers**

Use these exact public constants and functions:

```ts
export const ISSUE_MARKER = "<!-- repoguard-security-state -->";
export const ISSUE_TITLE = "RepoGuardAI: active critical/high security findings";

export function qualifyingFindings(findings: Finding[]): Finding[] {
  return findings.filter(
    (finding) =>
      finding.status !== "rejected" &&
      (finding.severity === "critical" || finding.severity === "high"),
  );
}

export function renderIssueBody(findings: Finding[], runUrl: string): string {
  const rows = findings.map((finding) => {
    const location = finding.file
      ? `${finding.file}${finding.line ? `:${finding.line}` : ""}`
      : "repository-wide";
    return `| ${finding.severity} | ${finding.id} | ${finding.title.replaceAll("|", "\\|")} | ${location.replaceAll("|", "\\|")} |`;
  });
  return [
    ISSUE_MARKER,
    "## Active RepoGuardAI findings",
    "",
    "| Severity | ID | Finding | Location |",
    "| --- | --- | --- | --- |",
    ...rows,
    "",
    `[Open the workflow run for complete evidence and artifacts.](${runUrl})`,
    "",
    "This issue is synchronized automatically. It never includes detected secret values or raw finding evidence.",
  ].join("\n");
}

export type IssueMutation =
  | { kind: "create"; title: string; body: string }
  | { kind: "update"; issueNumber: number; title: string; body: string }
  | { kind: "close"; issueNumber: number }
  | { kind: "noop" };

export function issueMutation(
  findings: Finding[],
  existing: { number: number; body: string } | null,
  runUrl = "",
): IssueMutation {
  const active = qualifyingFindings(findings);
  if (active.length === 0) {
    return existing ? { kind: "close", issueNumber: existing.number } : { kind: "noop" };
  }
  const body = renderIssueBody(active, runUrl);
  return existing
    ? { kind: "update", issueNumber: existing.number, title: ISSUE_TITLE, body }
    : { kind: "create", title: ISSUE_TITLE, body };
}
```

          name: repoguard-security-reports
